start with 1 
increment by 1 
nocache ;

  create table orders(
  onum number primary key,--주문번호
  aid varchar2(50) references accounts(id) on delete set null, --주문자id
  totprice number not null, --주문전체금액
  odate date default sysdate  --주문날짜
);

start with 1 
increment by 1 
nocache ;

  odnum number primary key, --상세주문번호
  aid varchar2(50),
  onum number references orders(onum) on delete cascade, --주문목록의 주문번호
  conum number references courses(conum) on delete set null, --강의번호
  oddate date default sysdate --주문테이블의 주문날짜와 같음
);
